# StyleClass

Add a StyleClass to Ogma. See StyleClassopen in new window

# Props

Prop nameDescriptionTypeValuesDefault
optionsThe Options to pass to the styleClass.object-
nodesThe NodeList to apply the class to. See referenceopen in new windowobject-() => (null)
edgesThe EdgeList to apply the class to. See referenceopen in new windowobject-() => (null)
<template>
  ...
  <StyleClass :options="classOptions" />
  ...
</template>
<script>
import { StyleClass } from "@linkurious/ogma-vue";
export default {
  name: "example",
  data() {
    return {
      classOptions: {
        name: "super class",
        nodes: this.ogma.getNodeList(),
        edges: this.ogma.getEdges(),
        nodeAttributes: {
          color: "blue"
        },
        edgeAttributes: {
          color: "blue"
        }
      }
    };
  }
};
</script>
