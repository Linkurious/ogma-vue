# NodeFilter

Creates an NodeFilter tranformation See NodeFilteropen in new window

# Props

Prop nameDescriptionTypeValuesDefault
eventsThe event you whish to register on.string-"all"
optionsThe options to pass to the node filter.object-{
criteria: () => true
}

# Events

Event namePropertiesDescription
enabledtransformation Object - The transformation objectTriggered when enabled. transformationEnabledopen in new window
disabledtransformation Object - The transformation objectTriggered when disabled. transformationDisabledopen in new window
refreshedtransformation Object - The transformation objectTriggered when refreshed. transformationRefreshedopen in new window
<template>
  ...
  <NodeFilter :options="filterOptions" @enabled="onEnabled" />
  ...
</template>
<script>
import { NodeFilter } from "@linkurious/ogma-vue";
export default {
  name: "example",
  data() {
    return {
      filterOptions: {
        enabled: true,
        criteria: () => true,
        duration: 1000
      }
    };
  },
  methods: {
    onEnabled(transformation) {
      console.log("Enabled: ", transformation);
    }
  }
};
</script>
